20template <rng::random_access_range Segments>

22class distributed_range_accessor {
23public:

44 operator=(const distributed_range_accessor &) noexcept = default;

46 constexpr distributed_range_accessor(Segments segments, size_type segment_id,
47 size_type idx) noexcept
48 : segments_(rng::views::all(std::forward<Segments>(segments))),
49 segment_id_(segment_id), idx_(idx) {}
50
51 constexpr distributed_range_accessor &
52 operator+=(difference_type offset) noexcept {
53
54 while (offset > 0) {
55 difference_type current_offset = std::min(
56 offset,
57 difference_type(rng::size(*(segments_.begin() + segment_id_))) -
58 difference_type(idx_));
59 idx_ += current_offset;
60 offset -= current_offset;
61
62 if (idx_ >= rng::size((*(segments_.begin() + segment_id_)))) {
63 segment_id_++;
64 idx_ = 0;
65 }
66 }
67
68 while (offset < 0) {
69 difference_type current_offset =
70 std::min(-offset, difference_type(idx_) + 1);
71
72 difference_type new_idx = difference_type(idx_) - current_offset;
73
74 if (new_idx < 0) {
75 segment_id_--;
76 new_idx = rng::size(*(segments_.begin() + segment_id_)) - 1;
77 }
78
79 idx_ = new_idx;
80 }
81
82 return *this;
83 }
84
85 constexpr bool operator==(const iterator_accessor &other) const noexcept {
86 return segment_id_ == other.segment_id_ && idx_ == other.idx_;
87 }
88
89 constexpr difference_type
90 operator-(const iterator_accessor &other) const noexcept {
91 return difference_type(get_global_idx()) - other.get_global_idx();
92 }
93
94 constexpr bool operator<(const iterator_accessor &other) const noexcept {
95 if (segment_id_ < other.segment_id_) {
96 return true;
97 } else if (segment_id_ == other.segment_id_) {
98 return idx_ < other.idx_;
99 } else {
100 return false;
101 }
102 }
103
104 constexpr reference operator*() const noexcept {
105 return *((*(segments_.begin() + segment_id_)).begin() + idx_);
106 }
107
108private:
109 size_type get_global_idx() const noexcept {
110 size_type cumulative_size = 0;
111 for (size_t i = 0; i < segment_id_; i++) {
112 cumulative_size += segments_[i].size();
113 }
114 return cumulative_size + idx_;
115 }
116
117 rng::views::all_t<Segments> segments_;
118 size_type segment_id_ = 0;
119 size_type idx_ = 0;
120};

183 segment_type tile(key_type tile_index) {
184 std::size_t tile_idx = tile_index[0] * grid_shape_[1] + tile_index[1];
185 auto values = values_[tile_idx].begin();
186 auto rowptr = rowptr_[tile_idx].begin();
187 auto colind = rowptr_[tile_idx].begin();

195 return segment_type(values, rowptr, colind, key_type{tm, tn}, nnz,

257 void init_() {
258 grid_shape_ = partition_->grid_shape(shape());
259 tile_shape_ = partition_->tile_shape(shape());
260
261 values_.reserve(grid_shape_[0] * grid_shape_[1]);
262 rowptr_.reserve(grid_shape_[0] * grid_shape_[1]);
263 colind_.reserve(grid_shape_[0] * grid_shape_[1]);
264 nnz_.reserve(grid_shape_[0] * grid_shape_[1]);
265
266 for (std::size_t i = 0; i < grid_shape_[0]; i++) {
267 for (std::size_t j = 0; j < grid_shape_[1]; j++) {
268 std::size_t rank = partition_->tile_rank(shape(), {i, j});
269
270 auto device = shp::devices()[rank];
271 shp::device_allocator<T> alloc(shp::context(), device);
272 shp::device_allocator<I> i_alloc(shp::context(), device);
273
274 values_.emplace_back(1, alloc, rank);
275 rowptr_.emplace_back(2, i_alloc, rank);
276 colind_.emplace_back(1, i_alloc, rank);
277 nnz_.push_back(0);
278 rowptr_.back()[0] = 0;
279 rowptr_.back()[1] = 0;
280 }
281 }
282 tiles_ = generate_tiles_();
283 segments_ = generate_segments_();

286 void init_random_(double density) {
287 grid_shape_ = partition_->grid_shape(shape());
288 tile_shape_ = partition_->tile_shape(shape());
289
290 values_.reserve(grid_shape_[0] * grid_shape_[1]);
291 rowptr_.reserve(grid_shape_[0] * grid_shape_[1]);
292 colind_.reserve(grid_shape_[0] * grid_shape_[1]);
293 nnz_.reserve(grid_shape_[0] * grid_shape_[1]);
294
295 for (std::size_t i = 0; i < grid_shape_[0]; i++) {
296 for (std::size_t j = 0; j < grid_shape_[1]; j++) {
297 std::size_t rank = partition_->tile_rank(shape(), {i, j});
298
299 size_t tm = std::min(tile_shape_[0], shape()[0] - i * tile_shape_[0]);
300 size_t tn = std::min(tile_shape_[1], shape()[1] - j * tile_shape_[1]);
301
302 auto device = shp::devices()[rank];
303 shp::device_allocator<T> alloc(shp::context(), device);
304 shp::device_allocator<I> i_alloc(shp::context(), device);
305
306 auto csr = generate_random_csr<T, I>({tm, tn});
307 std::size_t nnz = csr.size();
308
309 shp::device_vector<T, shp::device_allocator<T>> values(csr.size(),
310 alloc, rank);
311 shp::device_vector<I, shp::device_allocator<I>> rowptr(tm + 1, i_alloc,
312 rank);
313
314 shp::device_vector<I, shp::device_allocator<I>> colind(csr.size(),
315 i_alloc, rank);
316
317 shp::copy(csr.values_data(), csr.values_data() + csr.size(),
318 values.data());
319 shp::copy(csr.rowptr_data(), csr.rowptr_data() + tm + 1, rowptr.data());
320 shp::copy(csr.colind_data(), csr.colind_data() + csr.size(),
321 colind.data());
322
323 values_.push_back(std::move(values));
324 rowptr_.emplace_back(std::move(rowptr));
325 colind_.emplace_back(std::move(colind));
326 nnz_.push_back(nnz);
327 total_nnz_ += nnz;
328
329 delete[] csr.values_data();
330 delete[] csr.rowptr_data();
331 delete[] csr.colind_data();
332 }
333 }
334 tiles_ = generate_tiles_();
335 segments_ = generate_segments_();
336 }
