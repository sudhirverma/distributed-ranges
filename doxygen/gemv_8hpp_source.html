13template <lib::distributed_range C, typename T, typename I,
14 lib::distributed_range B>
15void gemv(C &&c, shp::sparse_matrix<T, I> &a, B &&b) {
16 namespace sycl = cl::sycl;
17
18 assert(c.size() == b.size());
19 assert(a.shape()[1] == b.size());
20 assert(a.grid_shape()[0] == c.segments().size());
21 assert(a.grid_shape()[1] == 1);
22
23 auto &&devices = shp::devices();
24
25 using b_scalar_type = std::ranges::range_value_t<B>;
26
27 using local_vector_type =
28 shp::device_vector<b_scalar_type, shp::device_allocator<b_scalar_type>>;
29
30 std::vector<local_vector_type> local_b;
31 std::vector<sycl::event> copy_events;
32 std::vector<sycl::event> comp_events;
33
34 for (std::size_t i = 0; i < devices.size(); i++) {
35 shp::device_allocator<T> allocator(shp::context(), devices[i]);
36 local_b.push_back(local_vector_type(b.size(), allocator, i));
37 }
38
39 for (auto &&l_b : local_b) {
40 auto event =
41 shp::copy_async(b.begin(), b.end(), lib::ranges::local(l_b.begin()));
42 copy_events.push_back(event);
43 }
44
45 for (size_t i = 0; i < a.grid_shape()[0]; i++) {
46 auto a_tile = a.tile({i, 0});
47
48 auto a_iter = a_tile.begin();
49 auto b_iter = lib::ranges::local(local_b[i].begin());
50 auto c_iter = lib::ranges::local(c.segments()[i].begin());
51
52 auto device = devices[a_tile.rank()];
53 sycl::queue q(device);
54
55 auto event = q.submit([=](auto &&h) {
56 h.depends_on(copy_events[i]);
57 h.parallel_for(sycl::range<1>(a_tile.size()), [=](sycl::id<1> idx) {
58 auto &&[index, a_v] = *(a_iter + idx);
59 auto &&[i, k] = index;
60 auto &&b_v = *(b_iter + k);
61 auto &&c_v = *(c_iter + i);
62 c_v += a_v * b_v;
63 });
64 });
65 comp_events.push_back(event);
66 }
67
68 for (auto &&event : comp_events) {
69 event.wait();
70 }
71}
