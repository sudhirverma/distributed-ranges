40 // Return the rank associated with a remote range.
41 // This is either:
42 // 1) r.rank(), if the remote range has a `rank()` method
43 // OR, if not available,
44 // 2) r.begin().rank(), if iterator is `remote_iterator`
45 template <rng::forward_range R>
46 requires((has_rank_method<R> && !disable_rank<std::remove_cv_t<R>>) ||
47 (has_rank_adl<R> && !disable_rank<std::remove_cv_t<R>>) ||
48 is_remote_iterator_shadow_impl_<rng::iterator_t<R>>)
49 constexpr auto operator()(R &&r) const {
50 if constexpr (has_rank_method<R> && !disable_rank<std::remove_cv_t<R>>) {
51 return std::forward<R>(r).rank();
52 } else if constexpr (is_remote_iterator_shadow_impl_<rng::iterator_t<R>>) {
53 return operator()(rng::begin(std::forward<R>(r)));
54 } else if constexpr (has_rank_adl<R> &&
55 !disable_rank<std::remove_cv_t<R>>) {
56 return rank_(std::forward<R>(r));
57 }
58 }
59
60 template <std::forward_iterator Iter>
61 requires(has_rank_method<Iter> && !disable_rank<std::remove_cv_t<Iter>>)
62 auto operator()(Iter iter) const {
63 if constexpr (has_rank_method<Iter> &&
64 !disable_rank<std::remove_cv_t<Iter>>) {
65 return std::forward<Iter>(iter).rank();
66 }
67 }
68};
