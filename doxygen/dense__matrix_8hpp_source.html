167 scalar_reference operator[](key_type index) {
168 std::size_t tile_i = index[0] / tile_shape_[0];
169 std::size_t tile_j = index[1] / tile_shape_[1];

184 std::size_t local_j = index[1] % tile_shape_[1];

198 std::vector<dense_matrix_view<
199 T, rng::iterator_t<shp::device_vector<T, shp::device_allocator<T>>>>>
200 tiles() {
201 std::vector<dense_matrix_view<
202 T, rng::iterator_t<shp::device_vector<T, shp::device_allocator<T>>>>>
203 views_;
204
205 for (size_t i = 0; i < grid_shape_[0]; i++) {
206 for (size_t j = 0; j < grid_shape_[1]; j++) {
207 auto iter = tiles_[i * grid_shape_[1] + j].begin();
208
209 size_t tm = std::min(tile_shape_[0], shape()[0] - i * tile_shape_[0]);
210 size_t tn = std::min(tile_shape_[1], shape()[1] - j * tile_shape_[1]);
211
212 views_.emplace_back(iter, key_type{tm, tn}, tile_shape_[1],
213 tiles_[i * grid_shape_[1] + j].rank());
214 }
215 }
216 return views_;
217 }

220 T, rng::iterator_t<shp::device_vector<T, shp::device_allocator<T>>>>>
221 segments() {

223 T, rng::iterator_t<shp::device_vector<T, shp::device_allocator<T>>>>>
224 views_;
225
226 for (size_t i = 0; i < grid_shape_[0]; i++) {
227 for (size_t j = 0; j < grid_shape_[1]; j++) {
228 auto iter = tiles_[i * grid_shape_[1] + j].begin();
229
230 size_t tm = std::min(tile_shape_[0], shape()[0] - i * tile_shape_[0]);
231 size_t tn = std::min(tile_shape_[1], shape()[1] - j * tile_shape_[1]);
232
233 size_t m_offset = i * tile_shape_[0];
234 size_t n_offset = j * tile_shape_[1];
235
236 views_.emplace_back(iter, key_type{tm, tn},
237 key_type{m_offset, n_offset}, tile_shape_[1],
238 tiles_[i * grid_shape_[1] + j].rank());
239 }
240 }
241 return views_;
242 }

244private:
245 void init_() {
246 grid_shape_ = partition_->grid_shape(shape());
247 tile_shape_ = partition_->tile_shape(shape());
248
249 tiles_.reserve(grid_shape_[0] * grid_shape_[1]);
250
251 for (std::size_t i = 0; i < grid_shape_[0]; i++) {
252 for (std::size_t j = 0; j < grid_shape_[1]; j++) {
253 std::size_t rank = partition_->tile_rank(shape(), {i, j});
254
255 auto device = shp::devices()[rank];
256 shp::device_allocator<T> alloc(shp::context(), device);
257
258 std::size_t tile_size = tile_shape_[0] * tile_shape_[1];
259
260 tiles_.emplace_back(tile_size, alloc, rank);
261 }
262 }
263 }
