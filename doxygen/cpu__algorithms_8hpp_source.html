36void copy(lib::distributed_contiguous_range auto &&in,
37 lib::distributed_iterator auto out) {
38 if (aligned(in.begin(), out)) {
39 for (const auto &&[in_seg, out_seg] :
40 rng::views::zip(local_segments(in), local_segments(out))) {
41 rng::copy(in_seg, out_seg.begin());
42 }
43 barrier();
44 } else {
45 lib::drlog.debug("copy: serial execution\n");
46 rng::copy(in, out);
47 fence();
48 }
49}

103template <lib::distributed_iterator DI, typename T>
104T reduce(int root, DI first, DI last, T init, auto &&binary_op) {
105 T result = 0;
106 auto comm = default_comm();
107
108 if (aligned(first)) {
109 lib::drlog.debug("Parallel reduce\n");
110 // reduce each segment, collect in a vector
111 std::vector<T> locals;
112 rng::for_each(
113 local_segments(rng::subrange(first, last)),
114 [&](auto v) { locals.push_back(v); },
115 [=](auto sr) {
116 return std::reduce(std::execution::par_unseq, sr.begin(), sr.end(),
117 T(0), binary_op);
118 });
119 // reduce the vector
120 auto local = std::reduce(std::execution::par_unseq, locals.begin(),
121 locals.end(), T(0), binary_op);
122
123 // Collect rank values in a vector
124 std::vector<T> all(comm.size());
125 comm.gather(local, all, root);
126 if (comm.rank() == root) {
127 result = std::reduce(std::execution::par_unseq, all.begin(), all.end(),
128 init, binary_op);
129 }
130 } else {
131 lib::drlog.debug("Serialreduce\n");
132 if (comm.rank() == root) {
133 // Reduce on root node
134 result =
135 std::reduce(std::execution::par_unseq, first, last, init, binary_op);
136 }
137 barrier();
138 }
139 return result;
140}

148void transform(lib::distributed_range auto &&in,
149 lib::distributed_iterator auto out, auto op) {
150 if (aligned(in.begin(), out)) {
151 for (const auto &&[in_seg, out_seg] :
152 rng::views::zip(local_segments(in), local_segments(out))) {
153 rng::transform(in_seg, out_seg.begin(), op);
154 }
155 barrier();
156 } else {
157 lib::drlog.debug("transform: serial execution\n");
158 rng::transform(in, out, op);
159 fence();
160 }
161}
